import { serve } from "https://deno.land/std@0.168.0/http/server.ts"

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders })

  try {
    const { memory } = await req.json()
    const GROQ_KEY = Deno.env.get('GROQ_KEY')

    const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
      method: 'POST',
      headers: { 
        'Authorization': `Bearer ${GROQ_KEY}`, 
        'Content-Type': 'application/json' 
      },
      body: JSON.stringify({
        model: "llama-3.1-8b-instant",
        messages: [
          { 
            role: "system", 
            content: "Eres un humorista español experto en chistes malos y juegos de palabras. Genera un chiste corto que sea gramaticalmente perfecto. No saludes, no expliques el chiste, solo devuelve el texto. Evita traducciones literales del inglés." 
          },
          { role: "user", content: "Genera un chiste original. No repitas estos temas: " + memory }
        ],
        temperature: 0.7
      })
    })
    
    const data = await response.json()
    const joke = data.choices[0].message.content.trim().replace(/^"|"$/g, '')

    return new Response(JSON.stringify({ joke }), { 
      headers: { ...corsHeaders, "Content-Type": "application/json" } 
    })
  } catch (err) {
    return new Response(JSON.stringify({ error: err.message }), { 
      status: 500, 
      headers: { ...corsHeaders, "Content-Type": "application/json" } 
    })
  }
})
