import { serve } from "https://deno.land/std@0.168.0/http/server.ts"

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders })

  try {
    const { memory } = await req.json()
    const GROQ_KEY = Deno.env.get('GROQ_KEY')

    // CAMBIO AL NUEVO MODELO GPT OSS 20B DE GROQ
    const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
      method: 'POST',
      headers: { 
        'Authorization': `Bearer ${GROQ_KEY}`, 
        'Content-Type': 'application/json' 
      },
      body: JSON.stringify({
        model: "gpt-oss-20b",
        messages: [
          { 
            role: "system", 
            content: "Eres un experto en humor español castizo y absurdo. Tu misión es generar chistes cortos que sean 'tan malos que son buenos'. Usa el razonamiento para que el juego de palabras tenga sentido en español de España. Formatos clásicos: ¿Qué le dice...?, Jaimito, Van dos y se cae el del medio, etc. Solo devuelve el texto del chiste, sin comentarios." 
          },
          { role: "user", content: "Genera un chiste único. No repitas nada parecido a esto: " + memory }
        ],
        temperature: 0.8 // Más alto para que sea más creativo y variado
      })
    })
    
    const data = await response.json()
    const joke = data.choices[0].message.content.trim().replace(/^"|"$/g, '')

    return new Response(JSON.stringify({ joke }), { 
      headers: { ...corsHeaders, "Content-Type": "application/json" } 
    })
  } catch (err) {
    return new Response(JSON.stringify({ error: err.message }), { 
      status: 500, 
      headers: { ...corsHeaders, "Content-Type": "application/json" } 
    })
  }
})
