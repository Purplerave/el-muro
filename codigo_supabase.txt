import { serve } from "https://deno.land/std@0.168.0/http/server.ts"

const corsHeaders = {
  'Access-Control-Allow-Origin': '*',
  'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type',
}

serve(async (req) => {
  if (req.method === 'OPTIONS') return new Response('ok', { headers: corsHeaders })

  try {
    const { memory } = await req.json()
    const GROQ_KEY = Deno.env.get('GROQ_KEY')

    const response = await fetch('https://api.groq.com/openai/v1/chat/completions', {
      method: 'POST',
      headers: { 
        'Authorization': `Bearer ${GROQ_KEY}`, 
        'Content-Type': 'application/json' 
      },
      body: JSON.stringify({
        model: "openai/gpt-oss-20b",
        messages: [
          { 
            role: "system", 
            content: "Eres un archivo de chistes CORTÍSIMOS españoles. Tu misión es dar un chiste de UNA SOLA LÍNEA o pregunta/respuesta. Máximo 15 palabras. Estilo: ¿Qué le dice...?, Jaimito, ¿Cómo se dice...? Ejemplo: '¿Qué hace una abeja en el gimnasio? Zumba.' NO INVENTES historias largas. Solo devuelve el texto del chiste corto seleccionado de tu memoria." 
          },
          { role: "user", content: "Dime un chiste clásico español cortísimo que no sea ninguno de estos: " + memory }
        ],
        temperature: 0.1, 
        top_p: 0.1,
        reasoning_effort: "low",
        stream: false
      })
    })
    
    const data = await response.json()
    const choice = data.choices && data.choices[0];
    let joke = (choice && choice.message) ? (choice.message.content || choice.message.reasoning_content || "") : "";
    joke = joke.trim().replace(/^"|"$/g, '');

    return new Response(JSON.stringify({ joke }), { 
      headers: { ...corsHeaders, "Content-Type": "application/json" } 
    })
  } catch (err) {
    return new Response(JSON.stringify({ error: err.message }), { 
      status: 500, 
      headers: { ...corsHeaders, "Content-Type": "application/json" } 
    })
  }
})
